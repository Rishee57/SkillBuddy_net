using System.Data;

namespace SkillBuddy.Entity
{


    public class EntityBase : IEntityBase
    {
        public Guid Id { get; set; }
        public bool Active { get; set; }
        public DateTime CreatedOn { get; set; }
        public Guid CreatedBy { get; set; }
        public string? CreatedByName { get; set; }
        public DateTime UpdatedOn { get; set; }
        public Guid UpdatedBy { get; set; }
        public string? UpdatedByName { get; set; }

        public EntityBase()
        {
            Id = Guid.NewGuid();
            Active = true;
            CreatedBy = UpdatedBy = new Guid();
            CreatedOn = UpdatedOn = DateTime.Now;
        }


        public void ParseDataRow(
            DataRow dRow,
            string colId = "Id",
            string colActive = "Active",
            string colCreatedOn = "CreatedOn",
            string colCreatedBy = "CreatedBy",
            string colCreatedByName = "CreatedByName",
            string colUpdatedOn = "UpdatedOn",
            string colUpdatedBy = "UpdatedBy",
            string colUpdatedByName = "UpdatedByName"

        )
        {
            if (ReferenceEquals(dRow, null))
                return;
            try
            {
                if (!String.IsNullOrEmpty(colId) && dRow.Table.Columns.Contains(colId))
                    Id = dRow.Field<Guid>(colId);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colActive) && dRow.Table.Columns.Contains(colActive))
                    Active = dRow.Field<Boolean>(colActive);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colCreatedOn) && dRow.Table.Columns.Contains(colCreatedOn))
                    CreatedOn = dRow.Field<DateTime>(colCreatedOn);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colCreatedBy) && dRow.Table.Columns.Contains(colCreatedBy))
                    CreatedBy = dRow.Field<Guid>(colCreatedBy);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colCreatedByName) && dRow.Table.Columns.Contains(colCreatedByName))
                    CreatedByName = dRow.Field<string>(colCreatedByName);
            }
            catch (Exception ex) { }

            try
            {
                if (!String.IsNullOrEmpty(colUpdatedOn) && dRow.Table.Columns.Contains(colUpdatedOn))
                    UpdatedOn = dRow.Field<DateTime>(colUpdatedOn);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colUpdatedBy) && dRow.Table.Columns.Contains(colUpdatedBy))
                    UpdatedBy = dRow.Field<Guid>(colUpdatedBy);
            }
            catch (Exception ex) { }
            try
            {
                if (!String.IsNullOrEmpty(colUpdatedByName) && dRow.Table.Columns.Contains(colUpdatedByName))
                    UpdatedByName = dRow.Field<string>(colUpdatedByName);
            }
            catch (Exception ex) { }
        }
    }
}
